"""
Temperature demo — Anthropic and OpenAI side by side
=====================================================
Key insight: temperature shows up on CREATIVE tasks, not factual ones.
Run the same prompt multiple times to see the variance.

temp = 0.0   →  identical every time (deterministic)
temp = 0.7   →  small variations 
temp = 1.5   →  real creativity (different valid names)
temp = 2.0   →  TOTAL CHAOS (model picks random tokens, breaks)

1. Temperature ranges DIFFER:
   Anthropic 0.0–1.0   (safer ceiling)
   OpenAI    0.0–2.0   (can break at the top)

2. Higher temperature = more random token selection
   too high = the model produces gibberish
"""
from dotenv import load_dotenv
load_dotenv()
import anthropic
import openai

anthropic_client = anthropic.Anthropic()
openai_client    = openai.OpenAI()

# A creative prompt — temperature has room to show its effect here
PROMPT = "Give me a name for a coffee shop. Just the name, nothing else."


def anthropic_ask(temp):
    r = anthropic_client.messages.create(
        model="claude-haiku-4-5",
        max_tokens=20,
        temperature=temp,
        messages=[{"role": "user", "content": PROMPT}],
    )
    return r.content[0].text.strip()


def openai_ask(temp):
    r = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        max_tokens=20,
        temperature=temp,
        messages=[{"role": "user", "content": PROMPT}],
    )
    return r.choices[0].message.content.strip()


print("=" * 60)
print("ANTHROPIC  — temperature range 0.0 to 1.0")
print("=" * 60)
for temp in [0.0, 0.7, 1.0]:
    print(f"\ntemp = {temp}  (run 3 times):")
    for i in range(3):
        print(f"   {anthropic_ask(temp)}")

print("\n" + "=" * 60)
print("OPENAI  — temperature range 0.0 to 2.0")
print("=" * 60)
for temp in [0.0, 0.7, 1.5, 2.0]:
    print(f"\ntemp = {temp}  (run 3 times):")
    for i in range(3):
        print(f"   {openai_ask(temp)}")
