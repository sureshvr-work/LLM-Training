# Build a big payload — repeat a sentence to fill tokens
BIGPROMPT=$(python3 -c "print('Tell me about banking. ' * 500)")

curl -s https://api.openai.com/v1/chat/completions \
  -H "Content-Type: application/json" \
    -H "Authorization: Bearer sk-proj-..." \
  -d "{
    \"model\": \"gpt-4o-mini\",
    \"messages\": [{\"role\": \"user\", \"content\": \"$BIGPROMPT\"}],
    \"max_tokens\": 5
  }" | python3 -c "
import sys, json
d = json.load(sys.stdin)
u = d.get('usage', {})
print(f'prompt_tokens={u.get(\"prompt_tokens\")} completion_tokens={u.get(\"completion_tokens\")}')
print(f'status: {d.get(\"object\", d.get(\"error\", {}).get(\"type\"))}')
"
